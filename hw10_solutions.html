def get_statistics(numbers: list[int]) -> dict:
return {
"sum": sum(numbers),
"max": max(numbers),
"min": min(numbers),
"average": sum(numbers) / len(numbers),
"count": len(numbers)
}

numbers = [10, 20, 30, 40, 50]
statistics = get_statistics(numbers)
print(statistics)

שימוש ב- filter ו-map עם רשימת מדינות:

countries = [
{'name': 'Israel', 'population': 9.3, 'birth': 1948},
{'name': 'United States', 'population': 331.9, 'birth': 1776},
{'name': 'Japan', 'population': 125.8, 'birth': 660},
{'name': 'Australia', 'population': 25.7, 'birth': 1901},
{'name': 'Canada', 'population': 38.0, 'birth': 1867}
]

# 1.שימוש ב-filter כדי לסנן מדינות עם יותר מ-30 מיליון תושבים


large_population_countries = list(filter(lambdacountry: country['population'] > 30,countries))

# 2. שימוש ב-filter כדי לסנן מדינות שהוקמו אחרי שנת 1800


post_1800_countries = list(filter(lambda country: country['birth'] > 1800, countries))

# 3. שימוש ב-map כדי ליצור רשימה של שמות המדינות

country_names = list(map(lambda country: country['name'], countries))
